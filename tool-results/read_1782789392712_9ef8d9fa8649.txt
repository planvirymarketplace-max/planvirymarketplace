   774→create view question_and_answer_views
   775→        (question_id, event_id, belongs_to, question_type, first_name, last_name, attendee_id, order_id, title,
   776→         answer)
   777→as
   778→SELECT q.id   AS question_id,
   779→       q.event_id,
   780→       q.belongs_to,
   781→       q.type AS question_type,
   782→       a.first_name,
   783→       a.last_name,
   784→       a.id   AS attendee_id,
   785→       qa.order_id,
   786→       q.title,
   787→       qa.answer,
   788→       qa.id as question_answer_id
   789→FROM question_answers qa
   790→         LEFT JOIN attendees a ON a.id = qa.attendee_id
   791→         JOIN orders o ON qa.order_id = o.id
   792→         JOIN questions q ON q.id = qa.question_id;
